98  {
99  $readAt = Carbon::now();
100  $markAs = ($request->asRead) ? 'read' : 'unread';
101 
102  if (isset($request->multipleMark) && is_array($request->multipleMark) && count($request->multipleMark) > 0) {
103  $notifications = auth()->user()->notifications()->whereIn('id', $request->multipleMark);
104  if ($markAs === 'unread') {
105  $notifications = $notifications->whereNotNull('read_at')->get();
106  $readAt = null;
107  } else {
108  $notifications = $notifications->whereNull('read_at')->get();
109  }
110  foreach ($notifications as $notification) {
111  $notification->update([
112  'read_at' => $readAt
113  ]);
114  }
115 
116  return response()->json(['result' => true], 200);
117  }
118 
119  $notification = auth()->user()->notifications()->find($request->notifyId);
120 
121  $notification->update([
122  'read_at' => ($notification && $markAs === 'read') ? $readAt : null
123  ]);
124 
125  return response()->json([
126  'result' => true, 'markAs' => $markAs, 'notifications' => auth()->user()->unreadNotifications
127  ], 200);
128  }
