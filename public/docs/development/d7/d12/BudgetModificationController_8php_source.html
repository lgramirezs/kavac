105

107  if (!is_null($request->type)) {

111  break;
112  case 'RE':
113  $codeFilter = 'budget.reductions';
114  break;

116  $codeFilter = 'budget.transfers';

142  );
143 
144  DB::transaction(function () use ($request, $code, $documentStatus) {

160  /** @var object Obtiene la formulación correspondiente a la acción específica seleccionada */

165  if ($formulation) {

185  'operation' => 'I',
186  'budget_sub_specific_formulation_id' => $formulation_transfer->id,
187  'budget_account_id' => $account['to_account_id'],
188  'budget_modification_id' => $budgetModification->id
189  ]);
190  }
191  }
192  }
