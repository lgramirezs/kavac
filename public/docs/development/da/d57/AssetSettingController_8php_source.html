82  list($table, $field) = explode("_", $key);

84 
85  if ($table === "asignations") {

91  } elseif ($table === "requests") {
92  /** @var string Define el modelo para asociado a las solicitudes de bienes */
93  $model = \Modules\Asset\Models\AssetRequest::class;
94  } elseif ($table === "reports") {
95  /** @var string Define el modelo para asociado a los reportes de bienes */
96  $model = \Modules\Asset\Models\AssetReport::class;
97  } elseif ($table === "inventories") {
98  /** @var string Define el modelo para asociado al inventario de bienes */
99  $model = \Modules\Asset\Models\AssetInventory::class;
100  }
101 
102  CodeSetting::updateOrCreate([
103  'module' => 'asset',
104  'table' => 'asset_'. $table,
105  'field' => $field,
106  ], [
107  'format_prefix' => $prefix,
108  'format_digits' => $digits,
109  'format_year' => $sufix,
110  'model' => $model,
111  ]);
112 
113  /** @var boolean Define el estatus verdadero para indicar que se ha registrado información */
114  $saved = true;
115  }
116  }
117 
118  if ($saved) {

120  } else {

123  'text' => 'No se registro ningún cambio'
124  ]);
125  }
