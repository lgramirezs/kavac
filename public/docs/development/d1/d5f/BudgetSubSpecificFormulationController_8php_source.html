347  public function importFormulation()
348  {
349  $headings = (new HeadingRowImport)->toArray(request()->file('file'));
350  $records = Excel::toArray(new DataImport, request()->file('file'))[0];
351  $msg = '';
352 
353  if (count($headings) < 1 || $headings[0] < 1) {
354  $msg = 'El archivo no contiene las cabeceras de los datos a importar.';
355  } elseif (count($headings) === 1 && $headings[0] >= 1) {
356  $validHeads = [
357  'codigo', 'total_real', 'total_estimado', 'total_anho', 'ene', 'feb', 'mar',
358  'abr', 'may', 'jun', 'jul', 'ago', 'sep', 'oct', 'nov', 'dic'
359  ];
360  foreach ($validHeads as $vh) {
361  if (!in_array($vh, $headings[0][0])) {
362  $msg = "El archivo no contiene una de las cabeceras requeridas.";
363  break;
364  }
365  }
366  } elseif (count($records) < 1) {
367  $msg = "El archivo no contiene registros a ser importados.";
368  }
369 
370  if (!empty($msg)) {
371  return response()->json(['result' => false, 'message' => $msg], 200);
372  }
373 
374  return response()->json(['result' => true, 'records' => $records], 200);
