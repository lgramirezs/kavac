583  *
584  * @return Array Listado de los registros a mostrar
585  */
586  public function getPayrollParameters(Request $request)
587  {
588  if (is_null($request->payroll_concepts)) {
589  $listGlobalParameters = [['id' => '', 'text' => 'Seleccione...']];
590  /**
591  * Objeto asociado al modelo Parameter
592  * @var Object $parameter
593  */
594  $parameters = Parameter::where(
595  [
596  'required_by' => 'payroll',
597  'active' => true,
598  ]
599  )->where('p_key', 'like', 'global_parameter_%')->get();
600 
601  if (!is_null($parameters)) {
602  foreach ($parameters as $parameter) {
603  $param = json_decode($parameter->p_value);
604  array_push($listGlobalParameters, [
605  'id' => $param->id,
606  'text' => $param->name,
607  ]);
608  }
609  }
610  return $listGlobalParameters;
611  } else {
612  $payrollParameters = [];
613  foreach ($request->payroll_concepts as $payroll_concept) {
614  $payrollConcept = PayrollConcept::find($payroll_concept['id']);
615  if ($payrollConcept && $payrollConcept->calculation_way == 'formula') {
616  $exploded = multiexplode(
617  [
618  'if', '(', ')', '{', '}', ' ',
619  '==', '<=', '>=', '<', '>', '!=',
620  '+', '-','*','/'
621  ],
622  $payrollConcept->formula
623  );
624  foreach ($exploded as $explod) {
625  /**
626  * Objeto asociado al modelo Parameter
627  * @var Object $parameters
628  */
629  $parameters = Parameter::where(
630  [
631  'required_by' => 'payroll',
632  'active' => true,
633  ]
634  )->where('p_value', 'like', '%' . $explod . '%')->get();
635  if ($parameters) {
636  foreach ($parameters as $parameter) {
637  $jsonValue = json_decode($parameter->p_value);
638  if (isset($jsonValue->id)) {
639  if ($jsonValue->id == $explod) {
640  if ($jsonValue->parameter_type == 'global_value') {
641  /** Si el parámetro es de valor global */
642  array_push($payrollParameters, [
643  'id' => $jsonValue->id,
644  'value' => $jsonValue->value
645  ]);
646  } elseif ($jsonValue->parameter_type == 'resettable_variable') {
647  /** Si el parámetro es reiniciable a cero por período de nómina */
648  array_push($payrollParameters, [
649  'id' => $jsonValue->id,
650  'value' => ''
651  ]);
652  } elseif ($jsonValue->parameter_type == 'processed_variable') {
653  /** Si el parámetro es una variable procesada */
654  array_push($payrollParameters, [
655  'id' => $jsonValue->id,
656  'value' => $jsonValue->formula
657  ]);
658  }
659  }
660  }
661  }
662  }
663  }
664  }
665  }
666  return $payrollParameters;
667  }
