46  {
47  $fields = [];
48  if ($type == 'exist') {
49  $fields = Asset::with('assetSpecificCategory')
50  ->select('asset_specific_category_id', \DB::raw('count(*) as total'))
51  ->groupBy('asset_specific_category_id');
52  $fields = ($order == 'desc')?$fields->orderBy('total', 'desc')->get():
53  $fields->orderBy('total', 'asc')->get();
54  } else {
55  $fields = \DB::table('asset_request_assets')
56  ->leftJoin('assets', 'asset_request_assets.asset_id', '=', 'assets.id')
57  ->leftJoin(
58  'asset_specific_categories',
59  'assets.asset_specific_category_id',
60  '=',
61  'asset_specific_categories.id'
62  )
63  ->select(
64  'asset_specific_categories.id',
65  'asset_specific_categories.name as name',
66  \DB::raw('count(*) as total')
67  )
68  ->groupBy('asset_specific_categories.id');
69 
70  $fields = ($order == 'desc')?$fields->orderBy('total', 'desc')->get():
71  $fields->orderBy('total', 'asc')->get();
72  }
73  return response()->json(['records' => $fields]);

129  $type_operation = 'requests';

131  }
132  $data_operations = \DB::table($table)
133  ->select('code', 'created_at')
134  ->get();
135  foreach ($data_operations as $data) {
136  array_push($fields, [
137  'code' => $data->code,
138  'type_operation' => $type_operation,
