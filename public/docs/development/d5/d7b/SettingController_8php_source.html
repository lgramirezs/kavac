169  $msgType = ['type' => 'store'];
170 
171  foreach ($parameters as $parameter) {
172  $parameterRepository->updateOrCreate(
173  ['p_key' => $parameter, 'required_by' => 'core'],
174  ['p_value' => (!is_null($request->$parameter)) ? 'true' : 'false']
175  );
176 
177  if ($parameter === "online") {
178  if (is_null($request->$parameter)) {
179  Artisan::call('up');
180  $msgType = [
181  'type' => 'other',
182  'text' => 'El sistema esta actualmente en línea, ' .
183  'todos los usuarios pueden acceder a la aplicación'
184  ];

186  $title = config('app.name') . ' - ' . __('En línea');

189  /** @var User Objeto con información de los usuarios del sistema */
190  $users = User::all();

192  $user->notify(new AppNotification($title, '', $description, true));
