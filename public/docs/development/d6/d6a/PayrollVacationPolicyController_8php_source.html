115  foreach ($assign_to as $field) {

117  $key = $field->id;
118  $options = [];

120  foreach ($payrollVacationPolicy->payrollConceptAssignOptions as $assign_option) {
121  if ($key == $assign_option['key']) {
122  if ($field->type == 'range') {
123  $options = json_decode($assign_option['value']);
124  } elseif ($field->type == 'list') {
125  $option = $field->model::find($assign_option['assignable_id']);
126  array_push(
127  $options,
128  [
129  'id' => $assign_option['assignable_id'],
130  'text' => $option->name
131  ]
132  );
133  }
134  }
135  }
136  $assign_options[$field->id] = $options;
137  }
138  }
139  }
140  }
141  $payrollVacationPolicy->assign_to = $assign_to;
142  $payrollVacationPolicy->assign_options = json_decode(json_encode($assign_options));
143  }
144 
145  return response()->json(
146  ['records' => $PayrollVacationPolicies],
147  200
